-- ============================================================
-- CBC SCHOOL MANAGEMENT SYSTEM — COMPLETE DATABASE SCHEMA
-- Target: Supabase (PostgreSQL)
-- Version: 1.0.0
-- ============================================================

-- ============================================================
-- SECTION 1A: EXTENSIONS
-- ============================================================
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- ============================================================
-- SECTION 1B: CUSTOM ENUM TYPES
-- ============================================================

-- User/staff status
CREATE TYPE user_status AS ENUM ('active', 'inactive', 'suspended', 'archived');

-- Student enrollment status
CREATE TYPE enrollment_status AS ENUM ('active', 'transferred', 'graduated', 'withdrawn', 'suspended');

-- Term definitions (CBC Kenya)
CREATE TYPE school_term AS ENUM ('Term 1', 'Term 2', 'Term 3');

-- Attendance status
CREATE TYPE attendance_status AS ENUM ('present', 'absent', 'late', 'excused');

-- Payment methods
CREATE TYPE payment_method AS ENUM ('cash', 'bank_transfer', 'mpesa', 'cheque', 'other');

-- Fee status
CREATE TYPE fee_status AS ENUM ('pending', 'partial', 'paid', 'overdue', 'waived');

-- Consent status
CREATE TYPE consent_status AS ENUM ('granted', 'denied', 'pending', 'withdrawn');

-- Message read status
CREATE TYPE read_status AS ENUM ('unread', 'read');

-- Notification types
CREATE TYPE notification_type AS ENUM ('info', 'warning', 'alert', 'success', 'system');

-- Audit action types
CREATE TYPE audit_action AS ENUM ('INSERT', 'UPDATE', 'DELETE', 'LOGIN', 'LOGOUT', 'FAILED_LOGIN', 'ROLE_CHANGE', 'CONFIG_CHANGE', 'PROMOTION', 'PAYMENT');

-- School type
CREATE TYPE school_type AS ENUM ('primary', 'junior_secondary', 'academy', 'mixed');

-- Gender
CREATE TYPE gender_type AS ENUM ('male', 'female', 'other');

-- Staff position
CREATE TYPE staff_position AS ENUM (
  'principal', 'deputy_principal', 'class_teacher', 'subject_teacher',
  'finance_officer', 'bursar', 'librarian', 'ict_admin', 'admin_staff', 'support_staff'
);

-- Discipline incident severity
CREATE TYPE incident_severity AS ENUM ('minor', 'moderate', 'major', 'critical');

-- CBC performance levels (4-point scale)
CREATE TYPE cbc_performance_level AS ENUM (
  'below_expectation',   -- 1
  'approaching',         -- 2
  'meeting',             -- 3
  'exceeding'            -- 4
);


-- ============================================================
-- SECTION 1C: SCHOOLS TABLE (Foundation — all tables reference this)
-- ============================================================
CREATE TABLE schools (
  school_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name VARCHAR(255) NOT NULL,
  type school_type NOT NULL DEFAULT 'primary',
  motto VARCHAR(255),
  address TEXT,
  county VARCHAR(100),
  sub_county VARCHAR(100),
  contact_email VARCHAR(255),
  contact_phone VARCHAR(20),
  logo_url TEXT,
  registration_number VARCHAR(100) UNIQUE,
  is_active BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON TABLE schools IS 'Core school profiles. All other tables reference school_id for RLS isolation.';


-- ============================================================
-- SECTION 1D: ROLES TABLE
-- ============================================================
CREATE TABLE roles (
  role_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name VARCHAR(50) NOT NULL UNIQUE,
  description TEXT,
  is_system_role BOOLEAN NOT NULL DEFAULT false,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON TABLE roles IS 'System roles for RBAC. Maps to 13 defined roles.';

-- Seed the 13 system roles
INSERT INTO roles (name, description, is_system_role) VALUES
  ('super_admin', 'System owner with full access across all schools', true),
  ('school_admin', 'School-level administrator with full school access', true),
  ('principal', 'School principal with academic and administrative oversight', true),
  ('deputy_principal', 'Deputy principal assisting the principal', true),
  ('teacher', 'General teacher role', true),
  ('class_teacher', 'Teacher assigned as class/homeroom teacher', true),
  ('subject_teacher', 'Teacher assigned to specific subjects', true),
  ('finance_officer', 'Handles fee structures, payments, and financial reports', true),
  ('parent', 'Parent/guardian with access to linked child data', true),
  ('student', 'Student with access to own records only', true),
  ('bursar', 'Handles day-to-day financial transactions', true),
  ('librarian', 'Manages library resources', true),
  ('ict_admin', 'Manages ICT infrastructure and system settings', true);


-- ============================================================
-- SECTION 1E: USERS TABLE
-- ============================================================
CREATE TABLE users (
  user_id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE RESTRICT,
  school_id UUID REFERENCES schools(school_id) ON DELETE RESTRICT,
  role_id UUID NOT NULL REFERENCES roles(role_id) ON DELETE RESTRICT,
  email VARCHAR(255) NOT NULL UNIQUE,
  first_name VARCHAR(100) NOT NULL,
  last_name VARCHAR(100) NOT NULL,
  middle_name VARCHAR(100),
  phone VARCHAR(20),
  gender gender_type,
  status user_status NOT NULL DEFAULT 'active',
  email_verified BOOLEAN NOT NULL DEFAULT false,
  failed_login_attempts INT NOT NULL DEFAULT 0,
  locked_until TIMESTAMPTZ,
  last_login_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_by UUID REFERENCES auth.users(id),
  updated_by UUID REFERENCES auth.users(id)
);

COMMENT ON TABLE users IS 'All system users. Links to Supabase auth.users. school_id is NULL for super_admin only.';
COMMENT ON COLUMN users.school_id IS 'NULL only for super_admin role. All others must belong to a school.';
COMMENT ON COLUMN users.failed_login_attempts IS 'Account locks after 5 failed attempts.';

-- Ensure non-super-admin users must have a school_id
-- (Enforced via application logic + trigger, since CHECK can't reference another table)


-- ============================================================
-- SECTION 1F: USER PROFILES (Extended info)
-- ============================================================
CREATE TABLE user_profiles (
  profile_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID NOT NULL UNIQUE REFERENCES users(user_id) ON DELETE RESTRICT,
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  date_of_birth DATE,
  address TEXT,
  photo_url TEXT,
  national_id VARCHAR(20),
  emergency_contact_name VARCHAR(200),
  emergency_contact_phone VARCHAR(20),
  blood_group VARCHAR(5),
  medical_conditions TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON TABLE user_profiles IS 'Extended profile data for all users. Sensitive fields protected by RLS.';

-- ============================================================
-- SECTION 2A: ACADEMIC YEARS
-- ============================================================
CREATE TABLE academic_years (
  academic_year_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  year VARCHAR(9) NOT NULL, -- e.g., '2025'
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  is_active BOOLEAN NOT NULL DEFAULT false,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_by UUID REFERENCES auth.users(id),

  CHECK (end_date > start_date)
);

COMMENT ON TABLE academic_years IS 'Academic year definitions per school. Only one can be active per school.';

-- Only one active academic year per school
CREATE UNIQUE INDEX idx_unique_active_academic_year
  ON academic_years (school_id) WHERE is_active = true;


-- ============================================================
-- SECTION 2B: TERMS
-- ============================================================
CREATE TABLE terms (
  term_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  name school_term NOT NULL,
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  is_active BOOLEAN NOT NULL DEFAULT false,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CHECK (end_date > start_date),
  UNIQUE (school_id, academic_year_id, name)
);

COMMENT ON TABLE terms IS 'Term definitions per academic year. Only one active term per school at a time.';

-- Only one active term per school
CREATE UNIQUE INDEX idx_unique_active_term
  ON terms (school_id) WHERE is_active = true;


-- ============================================================
-- SECTION 2C: GRADES / LEVELS (CBC Structure)
-- ============================================================
CREATE TABLE grades (
  grade_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  name VARCHAR(50) NOT NULL,          -- e.g., 'Grade 1', 'Grade 2', ... 'Grade 9'
  level_order INT NOT NULL,            -- 1, 2, 3... for sorting
  description TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (school_id, name),
  UNIQUE (school_id, level_order)
);

COMMENT ON TABLE grades IS 'CBC grade levels. Grade 1-6 (Primary), Grade 7-9 (Junior Secondary).';


-- ============================================================
-- SECTION 2D: CLASSES (Streams per grade)
-- ============================================================
CREATE TABLE classes (
  class_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  grade_id UUID NOT NULL REFERENCES grades(grade_id) ON DELETE RESTRICT,
  name VARCHAR(50) NOT NULL,           -- e.g., 'Grade 1 East', 'Grade 3 Blue'
  stream VARCHAR(50),                  -- e.g., 'East', 'West', 'Blue'
  capacity INT,
  class_teacher_id UUID REFERENCES users(user_id) ON DELETE SET NULL,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  is_active BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (school_id, grade_id, stream, academic_year_id),
  CHECK (capacity IS NULL OR capacity > 0)
);

COMMENT ON TABLE classes IS 'Class/stream definitions per grade per academic year. Each class can have one class teacher.';


-- ============================================================
-- SECTION 2E: LEARNING AREAS (CBC Subjects)
-- ============================================================
CREATE TABLE learning_areas (
  learning_area_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  name VARCHAR(150) NOT NULL,
  description TEXT,
  is_core BOOLEAN NOT NULL DEFAULT true,
  applicable_grades UUID[] DEFAULT '{}',  -- Array of grade_ids where this applies
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (school_id, name)
);

COMMENT ON TABLE learning_areas IS 'CBC learning areas (subjects). E.g., Mathematics, English, Kiswahili, Science & Technology, etc.';


-- ============================================================
-- SECTION 2F: STRANDS
-- ============================================================
CREATE TABLE strands (
  strand_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  learning_area_id UUID NOT NULL REFERENCES learning_areas(learning_area_id) ON DELETE RESTRICT,
  name VARCHAR(200) NOT NULL,
  description TEXT,
  sort_order INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (learning_area_id, name)
);

COMMENT ON TABLE strands IS 'Strands within each learning area. First level of CBC topic hierarchy.';


-- ============================================================
-- SECTION 2G: SUB-STRANDS
-- ============================================================
CREATE TABLE sub_strands (
  sub_strand_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  strand_id UUID NOT NULL REFERENCES strands(strand_id) ON DELETE RESTRICT,
  name VARCHAR(200) NOT NULL,
  description TEXT,
  sort_order INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (strand_id, name)
);

COMMENT ON TABLE sub_strands IS 'Sub-strands within each strand. Second level of CBC topic hierarchy.';


-- ============================================================
-- SECTION 2H: COMPETENCIES
-- ============================================================
CREATE TABLE competencies (
  competency_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  sub_strand_id UUID NOT NULL REFERENCES sub_strands(sub_strand_id) ON DELETE RESTRICT,
  name VARCHAR(255) NOT NULL,
  description TEXT,
  sort_order INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (sub_strand_id, name)
);

COMMENT ON TABLE competencies IS 'Specific competencies/skills under each sub-strand. Lowest level of CBC hierarchy. Assessment targets.';


-- ============================================================
-- SECTION 2I: PERFORMANCE LEVELS (CBC Rubric)
-- ============================================================
CREATE TABLE performance_levels (
  level_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  name VARCHAR(50) NOT NULL,
  label cbc_performance_level NOT NULL,
  numeric_value INT NOT NULL,
  description TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (school_id, label),
  CHECK (numeric_value >= 1 AND numeric_value <= 4)
);

COMMENT ON TABLE performance_levels IS 'CBC 4-point performance scale. 1=Below Expectation, 2=Approaching, 3=Meeting, 4=Exceeding.';

-- ============================================================
-- SECTION 3A: STUDENTS
-- ============================================================
CREATE TABLE students (
  student_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  user_id UUID UNIQUE REFERENCES users(user_id) ON DELETE RESTRICT,
  admission_number VARCHAR(50) NOT NULL,
  current_class_id UUID REFERENCES classes(class_id) ON DELETE SET NULL,
  date_of_birth DATE NOT NULL,
  gender gender_type NOT NULL,
  first_name VARCHAR(100) NOT NULL,
  last_name VARCHAR(100) NOT NULL,
  middle_name VARCHAR(100),
  enrollment_date DATE NOT NULL DEFAULT CURRENT_DATE,
  status enrollment_status NOT NULL DEFAULT 'active',
  photo_url TEXT,
  birth_certificate_no VARCHAR(50),
  nemis_number VARCHAR(50),
  has_special_needs BOOLEAN NOT NULL DEFAULT false,
  special_needs_details TEXT,
  medical_info TEXT,
  previous_school VARCHAR(255),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_by UUID REFERENCES auth.users(id),
  updated_by UUID REFERENCES auth.users(id),

  UNIQUE (school_id, admission_number)
);

COMMENT ON TABLE students IS 'Core student records. Links to user for login access. Special needs and medical info are RLS-protected.';
COMMENT ON COLUMN students.nemis_number IS 'National Education Management Information System number (Kenya).';


-- ============================================================
-- SECTION 3B: GUARDIANS (Parent/Guardian Profiles)
-- ============================================================
CREATE TABLE guardians (
  guardian_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  first_name VARCHAR(100) NOT NULL,
  last_name VARCHAR(100) NOT NULL,
  phone_number VARCHAR(20) NOT NULL,
  email VARCHAR(255),
  national_id VARCHAR(20),
  occupation VARCHAR(200),
  address TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_by UUID REFERENCES auth.users(id),
  updated_by UUID REFERENCES auth.users(id),

  UNIQUE (school_id, phone_number)
);

COMMENT ON TABLE guardians IS 'Standalone parent/guardian contact records (may or may not have a user account).';


-- ============================================================
-- SECTION 3C: STUDENT GUARDIANS (Parent-Student Link)
-- ============================================================
CREATE TABLE student_guardians (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  student_id UUID NOT NULL REFERENCES students(student_id) ON DELETE RESTRICT,
  guardian_id UUID REFERENCES guardians(guardian_id) ON DELETE RESTRICT,
  guardian_user_id UUID REFERENCES users(user_id) ON DELETE RESTRICT,
  relationship VARCHAR(50) NOT NULL,  -- e.g., 'father', 'mother', 'guardian'
  is_primary_contact BOOLEAN NOT NULL DEFAULT false,
  is_primary BOOLEAN NOT NULL DEFAULT false,
  can_pickup BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CHECK (guardian_id IS NOT NULL OR guardian_user_id IS NOT NULL),
  UNIQUE (student_id, guardian_id),
  UNIQUE (student_id, guardian_user_id)
);

COMMENT ON TABLE student_guardians IS 'Links parents/guardians to students. Supports both contact-only guardians and auth-linked parents.';


-- ============================================================
-- SECTION 3D: STUDENT CLASS HISTORY (Enrollment per term/year)
-- ============================================================
CREATE TABLE student_classes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  student_id UUID NOT NULL REFERENCES students(student_id) ON DELETE RESTRICT,
  class_id UUID NOT NULL REFERENCES classes(class_id) ON DELETE RESTRICT,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  term_id UUID NOT NULL REFERENCES terms(term_id) ON DELETE RESTRICT,
  status enrollment_status NOT NULL DEFAULT 'active',
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (student_id, academic_year_id, term_id)
);

COMMENT ON TABLE student_classes IS 'Student-to-class mapping per term. Enables tracking promotions and transfers.';


-- ============================================================
-- SECTION 3D: STAFF
-- ============================================================
CREATE TABLE staff (
  staff_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  user_id UUID NOT NULL UNIQUE REFERENCES users(user_id) ON DELETE RESTRICT,
  tsc_number VARCHAR(50),             -- Teachers Service Commission number
  position staff_position NOT NULL,
  employment_date DATE,
  contract_type VARCHAR(50),          -- 'permanent', 'contract', 'intern'
  qualification TEXT,
  status user_status NOT NULL DEFAULT 'active',
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_by UUID REFERENCES auth.users(id),

  UNIQUE (school_id, tsc_number)
);

COMMENT ON TABLE staff IS 'Staff records including TSC tracking. Links to users table.';
COMMENT ON COLUMN staff.tsc_number IS 'Teachers Service Commission registration number (Kenya). Nullable for non-teaching staff.';


-- ============================================================
-- SECTION 3E: TEACHER SUBJECT ASSIGNMENTS
-- ============================================================
CREATE TABLE teacher_subjects (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  teacher_id UUID NOT NULL REFERENCES staff(staff_id) ON DELETE RESTRICT,
  learning_area_id UUID NOT NULL REFERENCES learning_areas(learning_area_id) ON DELETE RESTRICT,
  class_id UUID NOT NULL REFERENCES classes(class_id) ON DELETE RESTRICT,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  term_id UUID NOT NULL REFERENCES terms(term_id) ON DELETE RESTRICT,
  is_active BOOLEAN NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (teacher_id, learning_area_id, class_id, academic_year_id, term_id)
);

COMMENT ON TABLE teacher_subjects IS 'Maps teachers to learning areas and classes per term. Controls assessment permissions.';


-- ============================================================
-- SECTION 3F: STAFF LEAVE MANAGEMENT
-- ============================================================
CREATE TABLE staff_leaves (
  leave_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  staff_id UUID NOT NULL REFERENCES staff(staff_id) ON DELETE RESTRICT,
  leave_type VARCHAR(50) NOT NULL,    -- 'sick', 'annual', 'maternity', 'paternity', 'compassionate'
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  reason TEXT,
  status VARCHAR(20) NOT NULL DEFAULT 'pending', -- 'pending', 'approved', 'rejected', 'cancelled'
  approved_by UUID REFERENCES users(user_id),
  approved_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CHECK (end_date >= start_date)
);

COMMENT ON TABLE staff_leaves IS 'Staff leave requests and approvals.';


-- ============================================================
-- SECTION 4A: ASSESSMENTS (Core CBC Assessment Records)
-- ============================================================
CREATE TABLE assessments (
  assessment_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  student_id UUID NOT NULL REFERENCES students(student_id) ON DELETE RESTRICT,
  competency_id UUID NOT NULL REFERENCES competencies(competency_id) ON DELETE RESTRICT,
  learning_area_id UUID NOT NULL REFERENCES learning_areas(learning_area_id) ON DELETE RESTRICT,
  class_id UUID NOT NULL REFERENCES classes(class_id) ON DELETE RESTRICT,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  term_id UUID NOT NULL REFERENCES terms(term_id) ON DELETE RESTRICT,
  score INT NOT NULL,
  level_id UUID NOT NULL REFERENCES performance_levels(level_id) ON DELETE RESTRICT,
  remarks TEXT,
  assessment_date DATE NOT NULL DEFAULT CURRENT_DATE,
  assessed_by UUID NOT NULL REFERENCES users(user_id) ON DELETE RESTRICT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_by UUID REFERENCES auth.users(id),

  -- Each student assessed once per competency per term
  UNIQUE (student_id, competency_id, term_id, academic_year_id),
  CHECK (score >= 1 AND score <= 4)
);

COMMENT ON TABLE assessments IS 'Individual CBC competency assessments. Score 1-4 maps to performance levels. Core data for report cards.';


-- ============================================================
-- SECTION 4B: ASSESSMENT AGGREGATES (Precomputed for performance)
-- ============================================================
CREATE TABLE assessment_aggregates (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  student_id UUID NOT NULL REFERENCES students(student_id) ON DELETE RESTRICT,
  learning_area_id UUID NOT NULL REFERENCES learning_areas(learning_area_id) ON DELETE RESTRICT,
  class_id UUID NOT NULL REFERENCES classes(class_id) ON DELETE RESTRICT,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  term_id UUID NOT NULL REFERENCES terms(term_id) ON DELETE RESTRICT,
  total_competencies INT NOT NULL DEFAULT 0,
  average_score DECIMAL(3,2) NOT NULL DEFAULT 0,
  overall_level cbc_performance_level,
  exceeding_count INT NOT NULL DEFAULT 0,
  meeting_count INT NOT NULL DEFAULT 0,
  approaching_count INT NOT NULL DEFAULT 0,
  below_expectation_count INT NOT NULL DEFAULT 0,
  computed_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  UNIQUE (student_id, learning_area_id, term_id, academic_year_id),
  CHECK (average_score >= 0 AND average_score <= 4),
  CHECK (total_competencies >= 0)
);

COMMENT ON TABLE assessment_aggregates IS 'Precomputed assessment summaries per student per learning area per term. Updated via trigger.';


-- ============================================================
-- SECTION 4C: REPORT CARDS
-- ============================================================
CREATE TABLE report_cards (
  report_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  student_id UUID NOT NULL REFERENCES students(student_id) ON DELETE RESTRICT,
  class_id UUID NOT NULL REFERENCES classes(class_id) ON DELETE RESTRICT,
  academic_year_id UUID NOT NULL REFERENCES academic_years(academic_year_id) ON DELETE RESTRICT,
  term_id UUID NOT NULL REFERENCES terms(term_id) ON DELETE RESTRICT,
  report_type VARCHAR(20) NOT NULL DEFAULT 'term',  -- 'term' or 'yearly'
  overall_average DECIMAL(3,2),
  overall_level cbc_performance_level,
  class_teacher_remarks TEXT,
  principal_remarks TEXT,
  analytics_json JSONB,              -- Detailed breakdown stored as JSON
  pdf_url TEXT,
  is_published BOOLEAN NOT NULL DEFAULT false,
  published_at TIMESTAMPTZ,
  generated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  generated_by UUID REFERENCES users(user_id),

  UNIQUE (student_id, term_id, academic_year_id, report_type),
  CHECK (report_type IN ('term', 'yearly')),
  CHECK (overall_average IS NULL OR (overall_average >= 0 AND overall_average <= 4))
);

COMMENT ON TABLE report_cards IS 'Generated report cards per student per term/year. Contains analytics JSON and optional PDF link.';


-- ============================================================
-- SECTION 4D: EXAM BANK
-- ============================================================
CREATE TABLE exam_bank (
  exam_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  school_id UUID NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
  learning_area_id UUID NOT NULL REFERENCES learning_areas(learning_area_id) ON DELETE RESTRICT,
  title VARCHAR(255) NOT NULL,
  description TEXT,
  content TEXT,
  exam_type VARCHAR(20) NOT NULL DEFAULT 'exam',
  file_url TEXT,
  file_name VARCHAR(255),
  file_type VARCHAR(100),
  term_id UUID REFERENCES terms(term_id) ON DELETE SET NULL,
  academic_year_id UUID REFERENCES academic_years(academic_year_id) ON DELETE SET NULL,
  created_by UUID REFERENCES users(user_id),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CHECK (exam_type IN ('exam', 'cat', 'mock', 'past_paper'))

-- SECTION 9A1: STUDENT GUARDIAN FLAG SYNC
-- Ensures is_primary and is_primary_contact stay consistent
-- ============================================================
CREATE OR REPLACE FUNCTION sync_student_guardian_primary_flags()
RETURNS TRIGGER AS $$
BEGIN
  IF COALESCE(NEW.is_primary, false) OR COALESCE(NEW.is_primary_contact, false) THEN
    NEW.is_primary := true;
    NEW.is_primary_contact := true;
  ELSE
    NEW.is_primary := false;
    NEW.is_primary_contact := false;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- ============================================================
-- SECTION 9B: AUDIT LOG TRIGGER FUNCTION
-- ============================================================
CREATE OR REPLACE FUNCTION fn_audit_log()
RETURNS TRIGGER AS $$
DECLARE
  v_action audit_action;
  v_old JSONB;
  v_new JSONB;
  v_school UUID;
BEGIN
  IF TG_OP = 'INSERT' THEN
    v_action := 'INSERT';
    v_old := NULL;
    v_new := to_jsonb(NEW);
    v_school := NEW.school_id;
  ELSIF TG_OP = 'UPDATE' THEN
    v_action := 'UPDATE';
    v_old := to_jsonb(OLD);
    v_new := to_jsonb(NEW);
    v_school := NEW.school_id;
  ELSIF TG_OP = 'DELETE' THEN
    v_action := 'DELETE';
    v_old := to_jsonb(OLD);
    v_new := NULL;
    v_school := OLD.school_id;
  END IF;

  INSERT INTO audit_logs (school_id, table_name, record_id, action, performed_by, old_data, new_data)
  VALUES (
    v_school,
    TG_TABLE_NAME,
    COALESCE(
      NEW.id,
      NEW.exam_id,
      NEW.exam_set_id,
      NEW.student_id,
      NEW.staff_id,
      NEW.user_id,
      NEW.guardian_id,
      OLD.id,
      OLD.exam_id,
      OLD.exam_set_id,
      OLD.student_id,
      OLD.staff_id,
      OLD.user_id,
      OLD.guardian_id
    ),
    v_action,
    COALESCE(current_setting('app.current_user_id', true)::UUID, NULL),
    v_old,
    v_new
  );

  RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- SECTION 9C: ENFORCE SINGLE ACTIVE ACADEMIC YEAR
-- ============================================================
CREATE OR REPLACE FUNCTION fn_enforce_single_active_year()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.is_active = true THEN
    UPDATE academic_years
    SET is_active = false, updated_at = NOW()
    WHERE school_id = NEW.school_id
      AND academic_year_id != NEW.academic_year_id
      AND is_active = true;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_enforce_single_active_year
  BEFORE INSERT OR UPDATE OF is_active ON academic_years
  FOR EACH ROW
  WHEN (NEW.is_active = true)
  EXECUTE FUNCTION fn_enforce_single_active_year();

-- ============================================================
-- SECTION 9E: AUTO-UPDATE PAYMENT TOTALS ON STUDENT FEES
-- ============================================================
CREATE OR REPLACE FUNCTION fn_update_student_fee_on_payment()
RETURNS TRIGGER AS $$
DECLARE
  v_total_paid DECIMAL(12,2);
  v_amount_due DECIMAL(12,2);
BEGIN
  -- Calculate total payments for this student_fee
  SELECT COALESCE(SUM(amount_paid), 0) INTO v_total_paid
  FROM payments
  WHERE student_fee_id = NEW.student_fee_id;

  -- Get amount due
  SELECT amount_due INTO v_amount_due
  FROM student_fees
  WHERE id = NEW.student_fee_id;

  -- Update student fee
  UPDATE student_fees
  SET
    amount_paid = v_total_paid,
    status = CASE
      WHEN v_total_paid >= v_amount_due THEN 'paid'
      WHEN v_total_paid > 0 THEN 'partial'
      ELSE 'pending'
    END,
    updated_at = NOW()
  WHERE id = NEW.student_fee_id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_fee_on_payment
  AFTER INSERT ON payments
  FOR EACH ROW EXECUTE FUNCTION fn_update_student_fee_on_payment();


-- ============================================================
-- SECTION 9F: AUTO-COMPUTE ASSESSMENT AGGREGATES
-- ============================================================
CREATE OR REPLACE FUNCTION fn_update_assessment_aggregate()
RETURNS TRIGGER AS $$
DECLARE
  v_student UUID;
  v_learning_area UUID;
  v_class UUID;
  v_term UUID;
  v_year UUID;
  v_school UUID;
  v_total INT;
  v_avg DECIMAL(3,2);
  v_exceeding INT;
  v_meeting INT;
  v_approaching INT;
  v_below INT;
  v_level cbc_performance_level;
BEGIN
  -- Get the relevant IDs
  IF TG_OP = 'DELETE' THEN
    v_student := OLD.student_id;
    v_learning_area := OLD.learning_area_id;
    v_class := OLD.class_id;
    v_term := OLD.term_id;
    v_year := OLD.academic_year_id;
    v_school := OLD.school_id;
  ELSE
    v_student := NEW.student_id;
    v_learning_area := NEW.learning_area_id;
    v_class := NEW.class_id;
    v_term := NEW.term_id;
    v_year := NEW.academic_year_id;
    v_school := NEW.school_id;
  END IF;

  -- Calculate aggregates
  SELECT
    COUNT(*),
    COALESCE(AVG(score), 0),
    COUNT(*) FILTER (WHERE score = 4),
    COUNT(*) FILTER (WHERE score = 3),
    COUNT(*) FILTER (WHERE score = 2),
    COUNT(*) FILTER (WHERE score = 1)
  INTO v_total, v_avg, v_exceeding, v_meeting, v_approaching, v_below
  FROM assessments
  WHERE student_id = v_student
    AND learning_area_id = v_learning_area
    AND term_id = v_term
    AND academic_year_id = v_year;

  -- Determine overall level
  v_level := CASE
    WHEN v_avg >= 3.5 THEN 'exceeding'
    WHEN v_avg >= 2.5 THEN 'meeting'
    WHEN v_avg >= 1.5 THEN 'approaching'
    ELSE 'below_expectation'
  END;

  -- Upsert aggregate
  INSERT INTO assessment_aggregates (
    school_id, student_id, learning_area_id, class_id,
    academic_year_id, term_id,
    total_competencies, average_score, overall_level,
    exceeding_count, meeting_count, approaching_count, below_expectation_count,
    computed_at
  ) VALUES (
    v_school, v_student, v_learning_area, v_class,
    v_year, v_term,
    v_total, v_avg, v_level,
    v_exceeding, v_meeting, v_approaching, v_below,
    NOW()
  )
  ON CONFLICT (student_id, learning_area_id, term_id, academic_year_id)
  DO UPDATE SET
    total_competencies = EXCLUDED.total_competencies,
    average_score = EXCLUDED.average_score,
    overall_level = EXCLUDED.overall_level,
    exceeding_count = EXCLUDED.exceeding_count,
    meeting_count = EXCLUDED.meeting_count,
    approaching_count = EXCLUDED.approaching_count,
    below_expectation_count = EXCLUDED.below_expectation_count,
    computed_at = NOW();

  RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_assessment_aggregate
  AFTER INSERT OR UPDATE OR DELETE ON assessments
  FOR EACH ROW EXECUTE FUNCTION fn_update_assessment_aggregate();

-- SECTION 9G: PREVENT DELETION OF REFERENCED USERS
-- ============================================================
CREATE OR REPLACE FUNCTION fn_prevent_user_delete()
RETURNS TRIGGER AS $$
BEGIN
  -- Check if user is referenced as a teacher, student, or staff
  IF EXISTS (SELECT 1 FROM staff WHERE user_id = OLD.user_id) THEN
    RAISE EXCEPTION 'Cannot delete user: linked to staff record. Deactivate instead.';
  END IF;
  IF EXISTS (SELECT 1 FROM students WHERE user_id = OLD.user_id) THEN
    RAISE EXCEPTION 'Cannot delete user: linked to student record. Deactivate instead.';
  END IF;
  IF EXISTS (SELECT 1 FROM student_guardians WHERE guardian_user_id = OLD.user_id) THEN
    RAISE EXCEPTION 'Cannot delete user: linked as guardian. Deactivate instead.';
  END IF;
  RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_prevent_user_delete
  BEFORE DELETE ON users
  FOR EACH ROW EXECUTE FUNCTION fn_prevent_user_delete();

-- SECTION 11A: GENERATE STUDENT ADMISSION NUMBER
-- ============================================================
CREATE OR REPLACE FUNCTION generate_admission_number(p_school_id UUID)
RETURNS TEXT AS $$
DECLARE
  v_year TEXT;
  v_count INT;
  v_prefix TEXT;
BEGIN
  v_year := EXTRACT(YEAR FROM CURRENT_DATE)::TEXT;

  SELECT COUNT(*) + 1 INTO v_count
  FROM students
  WHERE school_id = p_school_id
    AND EXTRACT(YEAR FROM enrollment_date) = EXTRACT(YEAR FROM CURRENT_DATE);

  v_prefix := 'ADM';
  RETURN v_prefix || '-' || v_year || '-' || LPAD(v_count::TEXT, 4, '0');
END;
$$ LANGUAGE plpgsql;

COMMENT ON FUNCTION generate_admission_number IS 'Generates sequential admission numbers like ADM-2025-0001';


-- ============================================================
-- SECTION 11B: CALCULATE TERM AVERAGE FOR A STUDENT
-- ============================================================
CREATE OR REPLACE FUNCTION calculate_term_average(
  p_student_id UUID,
  p_term_id UUID,
  p_academic_year_id UUID
)
RETURNS TABLE (
  learning_area_id UUID,
  learning_area_name VARCHAR,
  average_score DECIMAL,
  overall_level cbc_performance_level,
  total_competencies INT
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    la.learning_area_id,
    la.name,
    ROUND(AVG(a.score)::DECIMAL, 2),
    CASE
      WHEN AVG(a.score) >= 3.5 THEN 'exceeding'::cbc_performance_level
      WHEN AVG(a.score) >= 2.5 THEN 'meeting'::cbc_performance_level
      WHEN AVG(a.score) >= 1.5 THEN 'approaching'::cbc_performance_level
      ELSE 'below_expectation'::cbc_performance_level
    END,
    COUNT(a.assessment_id)::INT
  FROM assessments a
  JOIN learning_areas la ON a.learning_area_id = la.learning_area_id
  WHERE a.student_id = p_student_id
    AND a.term_id = p_term_id
    AND a.academic_year_id = p_academic_year_id
  GROUP BY la.learning_area_id, la.name;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;


-- ============================================================
-- SECTION 11C: GET STUDENT ATTENDANCE SUMMARY
-- ============================================================
CREATE OR REPLACE FUNCTION get_attendance_summary(
  p_student_id UUID,
  p_term_id UUID
)
RETURNS TABLE (
  total_days INT,
  present_days INT,
  absent_days INT,
  late_days INT,
  attendance_percentage DECIMAL
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    COUNT(*)::INT AS total_days,
    COUNT(*) FILTER (WHERE status = 'present')::INT,
    COUNT(*) FILTER (WHERE status = 'absent')::INT,
    COUNT(*) FILTER (WHERE status = 'late')::INT,
    ROUND(
      (COUNT(*) FILTER (WHERE status IN ('present', 'late'))::DECIMAL / NULLIF(COUNT(*), 0)) * 100,
      2
    )
  FROM attendance
  WHERE student_id = p_student_id
    AND term_id = p_term_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;


-- ============================================================
-- SECTION 11D: GET FEE BALANCE FOR A STUDENT
-- ============================================================
CREATE OR REPLACE FUNCTION get_student_fee_balance(
  p_student_id UUID,
  p_academic_year_id UUID
)
RETURNS TABLE (
  total_due DECIMAL,
  total_paid DECIMAL,
  total_balance DECIMAL
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    COALESCE(SUM(amount_due), 0)::DECIMAL,
    COALESCE(SUM(amount_paid), 0)::DECIMAL,
    COALESCE(SUM(amount_due - amount_paid), 0)::DECIMAL
  FROM student_fees
  WHERE student_id = p_student_id
    AND academic_year_id = p_academic_year_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Message recipients: users see own messages
CREATE POLICY msg_recipients_select ON message_recipients
  FOR SELECT USING (recipient_id = auth.uid());

CREATE POLICY msg_recipients_insert ON message_recipients
  FOR INSERT WITH CHECK (true); -- Controlled by service layer

CREATE POLICY msg_recipients_update ON message_recipients
  FOR UPDATE USING (recipient_id = auth.uid());
